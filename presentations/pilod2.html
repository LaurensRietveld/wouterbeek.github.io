[3.3] Metadata: endpoint

  PREFIX llo: 
  PREFIX xsd: 
  SELECT DISTINCT ((?dups / xsd:double(?dups + ?triples))
                  AS ?relDups) ?url
  WHERE {
    ?dataset llo:duplicates ?dups ;
      llo:triples ?triples ;
      llo:url ?url .
    FILTER(?triples > 0)
  }
  ORDER BY DESC(?relDups) LIMIT 50
